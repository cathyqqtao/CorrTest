## change working directory to the folder that contains data (branch length tree and RelTime timetree)
setwd('CorrTest//example')

## read outgroups
outgroups = read.table('groups.txt')
out.tip = character()

for (i in 1:length(outgroups[,1])){
  outgroup = sub("=outgroup","",outgroups[i,1])
  out.tip=c(out.tip, outgroup)
}


## read branch length tree and timetree
t.ml = read.tree('dosReis_Mammals274_ML.nwk')

t.rt = read.tree('dosReis_Mammals274_RelTime_oneCali_relTimes.nwk')

## run CorrTest
phylo.CorrTest(t.ml, t.rt, out.tip)
